heapq 란?

import heapq
 
힙 자료구조 상태로 만들어준다
 
import heapq

// 초기화
heap = []
heapq.heapify(heap)

// 자료 추가시 자동 오름차순으로 정렬구조를 갖는다

heap.heappush(heap,22) // [77]
heap.heappush(heap,77) // [22,77]
heap.heappush(heap,33) // [22,33,77]


//제거 후 리턴 
heap.heappop(heap) //22
heap //[33,77]
